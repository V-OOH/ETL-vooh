import pandas as pd
from datetime import datetime
import boto3
import os
from io import BytesIO, StringIO
from zoneinfo import ZoneInfo


hoje = datetime.now(ZoneInfo("America/Sao_Paulo")).strftime('%d_%m_%Y')

BUCKET = 'vooh-bucket'
PREFIXO_RAW = 'raw/'
PREFIXO_TRUSTED = 'trusted/'

s3 = boto3.client('s3')


def csvS3(prefixo_arquivo):
    resposta = s3.list_objects_v2(
        Bucket=BUCKET,
        Prefix=PREFIXO_RAW
    )

    arquivos = []

    for item in resposta.get('Contents', []):
        arquivo_s3 = item['Key']
        nomeArquivo = os.path.basename(arquivo_s3)

        if nomeArquivo.startswith(prefixo_arquivo) and hoje in nomeArquivo:
            arquivos.append(arquivo_s3)

    if not arquivos:
        raise FileNotFoundError(
            f'Nenhum arquivo {prefixo_arquivo} encontrado para {hoje}'
        )

    lista_csvs = []

    for arquivo in arquivos:
        objeto = s3.get_object(
            Bucket=BUCKET,
            Key=arquivo
        )

        conteudo = objeto['Body'].read()

        df = pd.read_csv(
            BytesIO(conteudo),
            encoding='utf-8',
            skipinitialspace=True
        )

        df.columns = df.columns.str.strip()
        lista_csvs.append(df)

    return pd.concat(lista_csvs, ignore_index=True)

#O script de captura salva os processos e os dados da máquinas em segundos diferentes, e isso retorna um erro na 
#hora de juntar os dois csv's lá na frente. Então trunquei para que valide somente a hora e minuto
def truncar_minuto(coluna):
    return pd.to_datetime(
        coluna,
        format='%d-%m-%Y %H:%M:%S'
    ).dt.strftime('%d-%m-%Y %H:%M')


def enviarCsv(df, nomeArquivo):
    buffer = StringIO()

    df.to_csv(
        buffer,
        index=False,
        encoding='utf-8'
    )

    s3.put_object(
        Bucket=BUCKET,
        Key=f'{PREFIXO_TRUSTED}{nomeArquivo}',
        Body=buffer.getvalue(),
        ContentType='text/csv'
    )
